(function(){ const params = new URLSearchParams(window.location.search); const tripId = params.get("tripId"); const token = params.get("token"); if(!tripId || !token){ document.body.innerHTML = "Invalid Link"; return; } let stops = []; /* ================= LOAD TRIP ================= */ async function loadTrip(){ try{ const res = await fetch( `/api/get-quote/add-stop/${tripId}?token=${token}` ); const data = await res.json(); if(!data.success){ document.body.innerHTML = data.message; return; } const t = data.trip; document.getElementById("tripInfo").innerHTML = `

Trip #: ${t.tripNumber}

Pickup: ${t.pickup}

Dropoff: ${t.dropoff}

`; }catch(err){ document.body.innerHTML = "Error loading trip"; } } /* ================= ADD ================= */ window.addStop = function(){ const input = document.getElementById("stopInput"); const val = input.value.trim(); if(!val) return; stops.push(val); input.value = ""; render(); } /* ================= RENDER ================= */ function render(){ const box = document.getElementById("stopsList"); box.innerHTML = stops.map((s,i)=>`
${i+1}. ${s}
`).join(""); } /* ================= CONFIRM ================= */ window.confirmStops = async function(){ if(!stops.length){ alert("Add stop first"); return; } try{ const res = await fetch( `/api/get-quote/add-stop/${tripId}/confirm`, { method:"POST", headers:{ "Content-Type":"application/json" }, body:JSON.stringify({ token, stops }) } ); const data = await res.json(); if(!data.success){ alert(data.message); return; } alert("Stops added successfully"); window.location.href = "/booking/index.html"; }catch(err){ alert("Error"); } } /* ================= CANCEL ================= */ window.goCancel = function(){ window.location.href = `/booking/cancel.html?token=${token}`; } /* ================= INIT ================= */ loadTrip(); })();
